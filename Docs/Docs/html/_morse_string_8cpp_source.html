153 void MorseString::toString(char * str, size_t n) const {
154  size_t usedSpace = 0;
155  memset(str, '\0', n);
156 
157  for (MorseChar const* mc : (*this)) {
158  size_t morseLen = strlen(mc->getMorseCode());
159  if ((usedSpace + morseLen) >= n)
160  return; // Do not overrun str
161 
162  // Append to str
163  char * _tstr = str + usedSpace;
164  strncpy_s(_tstr, n - usedSpace, mc->getMorseCode(), morseLen);
165 
166  // Always finish a letter with '/' unless we're printing a space.
167  if (mc->getMorseCode()[0] != ' ') {
168  _tstr = _tstr + morseLen;
169  strncpy_s(_tstr, 2, "/", 2);
170  ++morseLen;
171  }
172 
173  usedSpace += morseLen;
174  }
175 }

177 void MorseString::createFromAscii(MorseString & result, const char * ascii, size_t n) {
178  for (size_t i = 0; i < n; ++i) {
179  char _asc = ascii[i];
180  if (_asc >= 'a' && _asc <= 'z') _asc &= 0xFFDF; // to-upper-case
181 
182  MorseChar const* code = allCodes().getByAscii(_asc);
183  result.add(const_cast<MorseChar *>(code));
184  }
